#include <stdio.h>
#include
#include <stdlib.h>
#include<opencv2\core\core.hpp>
#include<opencv2\opencv.hpp>
#include<opencv2\highgui.hpp>
#include<opencv2/objdetect/objdetect.hpp>
#include <opencv2\imgproc\types_c.h>

using namespace std;
using namespace cv;

string xmlPath=”C:\Users\12238\Desktop\opencv\haarcascade_frontalface_default.xml”;//xml在F:\opencv\opencv\sources\data\haarcascades里复制或直接引用

int main(void)
{
cv::Mat a = cv::imread(“he.jpg”);
CascadeClassifier ccf; //创建脸部对象
ccf.load(xmlPath); //导入opencv自带检测的文件
vector faces;
Mat gray;
cvtColor(a,gray,CV_BGR2GRAY);
equalizeHist(gray,gray);
ccf.detectMultiScale(gray,faces,1.06,10);
cout<<”123453”<<endl;
for(vector::const_iterator iter=faces.begin();iter!=faces.end();iter++)
{
rectangle(a,*iter,Scalar(0,0,255),2,8); //画出脸部矩形
}
cv::imshow(“hello”,a);
cv::waitKey(0);
return 0;
}
